【leafletJs】基本使用

2024-04-16 2 min read # es6

安装

1. 下载leaflet

npm i leaflet

2. 入口文件引入leafletJs的样式

import "leaflet/dist/leaflet.css";

使用

1. 封装组件

基于vue封装的一个基础展示地图的组件 ：LeafletMap.vue

<script setup lang="ts">
import { ref, onMounted } from "vue";
import L from "leaflet";
interface Props {
  isGetLocation: boolean;
  width?: string;
  height?: string;
  latitude?: number;
  longitude?: number;
}
const props = withDefaults(defineProps<Props>(), {
  isGetLocation: false,
  width: "100vw",
  height: "100vh",
  latitude: 0,
  longitude: 0,
});
onMounted(async () => {
  if (props.latitude && props.longitude) {
    loadMap({
      latitude: props.latitude,
      longitude: props.longitude,
    });
    return;
  }
  if (props.isGetLocation) {
    const res = await getPostion();
    if (res.coords) {
      loadMap({
        latitude: res.coords.latitude,
        longitude: res.coords.longitude,
      });
      return;
    }
  }
  loadMap({});
});
function getPostion() {
  return new Promise((res, rej) => {
    if ("geolocation" in navigator) {
      /* geolocation 存在 */
      function success(position) {
        res(position);
      }

      function error() {
        rej("Sorry, no position available.");
        alert("Sorry, no position available.");
      }

      const options = {
        enableHighAccuracy: true,
        // maximumAge: 30000,
        // timeout: 27000,
      };
      const watchID = navigator.geolocation.watchPosition(
        success,
        error,
        options,
      );
    } else {
      return 0;
      /* geolocation 不存在 */
    }
  });
}
function loadMap({ latitude, longitude }) {
  // 创建实例
  var map = L.map("map").setView([latitude || 0, longitude || 0], 13);
  // 获取地图片
  L.tileLayer("http://{s}.tile.osm.org/{z}/{x}/{y}.png").addTo(map);
  // 显示定位图表
  L.marker([latitude || 0, longitude || 0])
    .addTo(map)
    // .bindPopup("marker")
    .openPopup();
}
</script>

<template>
  <div>
    <div :style="{ width: width, height: height }" id="map"></div>
  </div>
</template>
