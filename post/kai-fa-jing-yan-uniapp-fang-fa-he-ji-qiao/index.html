1.截屏保存到本地

        const toShare=()=>{       
                    var pages = getCurrentPages();
                    var page = pages[pages.length - 1];
                    var ws = page.$getAppWebview();
                    // console.log(currentWebview.id);//获得当前webview的id
                    // console.log(currentWebview.isVisible());//查询当前webview是否可见
                    var bitmap = null;
                    bitmap = new plus.nativeObj.Bitmap('test');
                    // 将webview内容绘制到Bitmap对象中
                    uni.showLoading({
                        // title: '加载中'
                    });
                    ws.draw(bitmap, function() {
                        var bitmaps = plus.nativeObj.Bitmap.getItems();;
                        bitmap.save("_doc/a.jpg", {}, function(i) {
                                uni.saveImageToPhotosAlbum({
                                    filePath: i.target,
                                    success: function() {
                                        bitmap.clear();
                                        setTimeout(() => {
                                            uni.hideLoading();
                                            uni.showToast({
                                                title: "截屏成功",
                                                icon: "none"
                                            });
                                        }, 2000)
                                    },
                                })
                            },
                            function(e) {
                                console.log('保存图片失败：' + JSON.stringify(e));
                            });
                    }, function(e) {
                        console.log('截屏绘制图片失败：' + JSON.stringify(e));
                    }, {
                        check: true, // 设置为检测白屏
                        clip: {
                            top: '0',
                            left: '0',
                            height: '100%',
                            width: '100%'
                        } //设置截屏区域
                    });
                }

					}
				})
                                if (!obj.value.length) {
						return
					}
				if (obj.label == 'phone') {
					window.location.href = `tel://${obj.value}`
				}
				if (obj.label == 'sms') {
					if (osName == 'ios') {
						window.location.href = `sms:/open?addresses=${obj.value}`
					} else {
						window.location.href = `sms:${obj.value}`
					}
				}
				if (obj.label == 'msg') {
					window.location.href = `mailto:${obj.value}`
				}

5.设置tabbar中间按钮

实现

在page.json找到tabBar配置项，使用tabBar的midButton属性配置。

	"tabBar": {
		"color": "#BBBBBB",
		"selectedColor": "#2F86F6",
		"borderStyle": "black",
		"backgroundColor": "#FFFFFF",
		"midButton":{  //这里是配置项
			"iconPath": "static/tabbar/mid.png",
			"iconWidth": "40px",
			"iconHeight":"75px",
			"text": ""
		},
		"list": [
      		{
      			"pagePath": "pages/wallet/wallet",
      			"iconPath": "static/tabbar/tab_wallet.png",
      			"selectedIconPath": "static/tabbar/user_1.png",
      			"text": ""
      		},
      		{
      			"pagePath": "pages/information/information",
      			"iconPath": "static/tabbar/tab_msg.png",
      			"selectedIconPath": "static/tabbar/user_1.png",
      			"text": ""
      		}
		]

midButton属性有一个缺点，不能配置path路径。此时有一个问题，我们不能在初始化时就显示中间midButton对应的页面。
解决方案：在tabBar.list属性中配置好index页面路径，然后使用visible属性将 index 页面的tabbar隐藏，此时只需要配置好midButton的icon图标就好了。


	"tabBar": {
		"color": "#BBBBBB",
		"selectedColor": "#2F86F6",
		"borderStyle": "black",
		"backgroundColor": "#FFFFFF",
		"midButton":{
			"iconPath": "static/tabbar/mid.png",
			"iconWidth": "40px",
			"iconHeight":"75px",
			"text": ""
		},
		"list": [
// @添加内容
			{
			"pagePath": "pages/index/index",
			"iconPath": "static/tabbar/ic_house.png",
			"selectedIconPath": "static/tabbar/ic_house_active.png",
			"text": "",
			"visible":false
		},
// 
		{
			"pagePath": "pages/wallet/wallet",
			"iconPath": "static/tabbar/tab_wallet.png",
			"selectedIconPath": "static/tabbar/user_1.png",
			"text": ""
		},
		{
			"pagePath": "pages/information/information",
			"iconPath": "static/tabbar/tab_msg.png",
			"selectedIconPath": "static/tabbar/user_1.png",
			"text": ""
		}
		]
	},

还需要配置一下初始页面，在app.vue中onLunch方法下配置

	onLaunch: function() {
		...
			//指定初始化页面
			uni.switchTab({
				url: `/pages/index/index`
			})
		},

最后在app.vue页面监听midButton按钮的点击事件，实现tabbar页面的切换

注意使用uni.switchTab()方法实现tabbar切换

		mounted() {
			uni.onTabBarMidButtonTap(() => {
				uni.switchTab({
					url: `/pages/index/index`
				})
			})
		},

总结

  1. 配置midButton属性
  2. 在tabBar.list属性中配置midButton对应的页面，并将visible属性设置成false
  3. 配置初始页面（可选）
  4. 监听midButton按钮的点击事件
