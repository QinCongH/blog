5. 配置路径别名

  1. 安装@types/node: 解决在ts环境中找不到node中间件的报错
npm i @types/node
  1. 在vite.config.ts中添加配置
import { defineConfig } from 'vite'
import {resolve} from 'path'
import react from '@vitejs/plugin-react'
// https://vitejs.dev/config/
export default defineConfig({
plugins: [react()],
resolve:{
alias:{
'@':resolve(__dirname,'./src')
}
}
})
  1. ts中配置寻找路径目录 tsconfig.json
"compilerOptions": {
...
"baseUrl": "./",
"paths": {
"@/*":["src/*"]
}
